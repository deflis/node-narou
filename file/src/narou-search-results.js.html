/**
 * なろう小説API検索結果
 * @class NarouSearchResults
 */
export default class {
    /**
     * @constractor
     * @private
     */
    constructor(result, params) {
        let count = result.shift().allcount;
        
        let limit = 20;
        if (params.hasOwnProperty("lim")) {
            limit = params.lim;
        }

        let start = 0;
        if (params.hasOwnProperty("start")) {
            start = params.start;
        }
        
        /**
         * 検索結果数
         * @type {number}
         */
        this.allcount = count;
        /**
         * 結果表示上限数
         * @type {number}
         */
        this.limit = limit;
        /**
         * 結果表示開始数
         * @type {number}
         */
        this.start = start;
        /**
         * 結果表示ページ数
         * @type {number}
         */
        this.page = start / limit;
        /**
         * 結果数
         * @type {number}
         */
        this.length = result.length;
        /**
         * 検索結果
         * @type {NarouSearchResult[]}
         */
        this.values = result;
    }
};
